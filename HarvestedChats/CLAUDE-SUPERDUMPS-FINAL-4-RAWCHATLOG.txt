  const SEEDS = 5;

  const STATUS = ['PASS','WARN','FAIL','EDGE_CASE','QUESTION','UPGRADE'];
  const CHECKPOINTS = ['HTML','CSS','JS','DOM','UI Anim','Charts','Gamification','Debug','Security','Performance','A11y','SEO'];
  const CATEGORIES = ['Integrity','Performance','Edge Cases','Error Handling','Visual','Interaction','Security','Upgrade'];

  // Generate full QA questions
  const QA_QUESTIONS = [];
  for(let i=1;i<=TOTAL_QA;i++){
    const checkpoint = CHECKPOINTS[i % CHECKPOINTS.length];
    const category = CATEGORIES[i % CATEGORIES.length];
    QA_QUESTIONS.push({ id:i, checkpoint, category, question:`Q${i}: ${checkpoint} / ${category}` });
  }

  // Generate audit points including devil advocate + risk
  const generateAuditPoint = (qa) => {
    const result = STATUS[Math.floor(Math.random()*STATUS.length)];
    const risk = (result==='FAIL'||result==='EDGE_CASE') ? 'High' : (result==='WARN') ? 'Medium' : 'Low';
    const devil_advocate = `Might be acceptable due to system tolerance: ${qa.checkpoint}/${qa.category}`;
    return { ...qa, result, risk, devil_advocate };
  };

  // Store condensed audit per seed
  const condensedAudit = {};
  for(let s=0;s<SEEDS;s++){
    const seedQA = QA_QUESTIONS.slice(s*(TOTAL_QA/SEEDS), (s+1)*(TOTAL_QA/SEEDS));
    condensedAudit[`SEED_${s+1}`] = seedQA.map(generateAuditPoint);

    try{
      const data = {
        timestamp: new Date().toISOString(),
        seeds: condensedAudit

      const blob = new Blob([JSON.stringify(data,null,2)],{type:'application/json'});

      const a = document.createElement('a');

      a.download = `genesis-audit-condensed-${Date.now()}.json`;

      console.error('Export failed:',e);
      alert('Audit export failed. Check console.');
