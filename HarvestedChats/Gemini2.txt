// This is the V1.1 Hardened API. It is stable, proven, and sacred.

// It is our "single source of truth" for system events.

const NET_SYSTEM_V1_1 = (() => {

    'use strict';

    let isReady = false;

    const callQueue = [];

    const subscribers = new Map();



    const log = (msg) => console.log(`[V1.1 Core] ${msg}`);

    const safeCall = (func) => { try { func(); } catch (e) { console.error(e); } };



    const publish = (eventName, data) => {

        if (subscribers.has(eventName)) {

            subscribers.get(eventName).forEach(cb => safeCall(() => cb(data)));

        }

    };



    const init = () => {

        if (isReady) return;

        isReady = true;

        log("System logic ready. Processing queue...");

        callQueue.forEach(f => safeCall(f));

        callQueue.length = 0;

        publish('ready');

    };



    const on = (eventName, callback) => {

        if (!subscribers.has(eventName)) {

            subscribers.set(eventName, new Set());

        }

        subscribers.get(eventName).add(callback);

        log(`New subscriber added for: ${eventName}`);

        // Return an 'unsubscribe' function

        return () => {

            if (subscribers.has(eventName)) {

                subscribers.get(eventName).delete(callback);

                log(`Subscriber removed for: ${eventName}`);

            }

        };

    };



    const execute = (func) => {

        if (isReady) {

            safeCall(func);

        } else {

            callQueue.push(func);

        }

    };



    // Public API

    return {

        // Core Methods

        init,

        on,      // The *only* way to listen for events

        execute, // The *only* way to queue an action

        publish  // The *only* way to fire an event

    };

})();
