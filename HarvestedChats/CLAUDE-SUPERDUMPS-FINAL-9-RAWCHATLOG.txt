(function() {
  const claudeState = window.claudeState || {
    executedSnippets: {},
    auditPoints: 0
  };

  // ---------- Register a new snippet ----------
  function registerClaudeSnippet(id, fn) {
    if (typeof fn !== 'function') throw new Error('Snippet must be a function');
    claudeState[id] = fn;
    console.log([Claude] Registered snippet: ${id});
  }

  // ---------- Execute a snippet by ID ----------
  async function executeClaudeSnippet(id, ...args) {
    if (!claudeState[id]) throw new Error(Snippet ${id} not found);
    try {
      const result = await claudeState[id](...args);
      claudeState.executedSnippets[id] = { result, timestamp: new Date().toISOString() };
      claudeState.auditPoints += 1; // Track audit points
      console.log([Claude] Executed snippet: ${id}, Audit points: ${claudeState.auditPoints});
      return result;
    } catch (err) {
      console.error([Claude] Error executing snippet ${id}:, err);
      throw err;
    }
  }

  // ---------- List all registered snippets ----------
  function listClaudeSnippets() {
    return Object.keys(claudeState).filter(k => typeof claudeState[k] === 'function');
  }

  // ---------- Expose to global ----------
  window.claudeState = claudeState;
  window.registerClaudeSnippet = registerClaudeSnippet;
  window.executeClaudeSnippet = executeClaudeSnippet;
  window.listClaudeSnippets = listClaudeSnippets;

  console.log('[Claude] Snippet registry initialized');
})();

class ClaudeSnippets {
    constructor(artifactUrl) {
        this.artifactUrl = artifactUrl;
        this.snippets = [];
        this.fetchInProgress = false;
    }

    log(msg) {
        console.log([ClaudeSnippets] ${msg});
    }

    async fetchArtifact() {
        this.log(Fetching artifact from ${this.artifactUrl}...);
        try {
            const response = await fetch(this.artifactUrl);
            if (!response.ok) throw new Error(HTTP ${response.status});
            const data = await response.json();
            this.log(Artifact fetched: ${data.snippets?.length || 0} existing snippets);
            if (data.snippets) this.snippets.push(...data.snippets);
        } catch (e) {
            this.log(Error fetching artifact: ${e.message});
        }
    }

    async requestAdditionalSnippets(count = 10) {
        if (this.fetchInProgress) return;
        this.fetchInProgress = true;
        this.log(Requesting ${count} additional snippets from Claude...);
        try {
            // NOTE: Replace this simulated fetch with actual Claude API call
            // when credentials or endpoint are available
            const simulatedResponse = Array.from({ length: count }, (_, i) => Snippet_${Date.now()}_${i});
            this.snippets.push(...simulatedResponse);
            this.log(Received ${simulatedResponse.length} new snippets);
        } catch (e) {
            this.log(Error requesting snippets: ${e.message});
        } finally {
            this.fetchInProgress = false;
        }
    }

    listSnippets() {
        return this.snippets;
    }

    async run(count = 10) {
        await this.fetchArtifact();
        await this.requestAdditionalSnippets(count);
        this.log(Total snippets available: ${this.snippets.length});
        return this.snippets;
    }
}
